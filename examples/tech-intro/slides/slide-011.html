export function applyPatchesByLine(
  html: string,
  patches: WysiwygPatch[]
): string {
  const lines = html.split('\n')

  for (const patch of patches) {
    // 以行号定位目标元素
    const i = patch.anchor.line - 1
    const line = lines[i]

    switch (patch.type) {
      case 'style-prop':
        // 精确替换 / 追加单个 CSS 属性
        lines[i] = setInlineStyleProp(
          line, patch.property, patch.value
        )
        break

      case 'text':
        // 只改标签内文本，不碰其他属性
        lines[i] = line.replace(
          /(>[^<]*)(<\/)/,
          `>${escaped}$2`
        )
        break

      case 'move':
        lines[i] = setInlineStyleProp(
          line, 'transform',
          `translate(${patch.dx}px,${patch.dy}px)`
        )
        break

      case 'delete':
        // 注释整个元素块
        lines.splice(i, end-i+1,
          `<!-- [tang-deleted]\n${block}\n-->`)
        break
    }
  }

  return cleanInjectAttrs(lines.join('\n'))
}
