#### 3.1.2 渲染引擎（Slide Runner）

```typescript
// packages/core/src/runner/SlideRunner.ts

interface SlideRunnerOptions {
  container: HTMLElement
  slidesDir: string        // slides/ 目录路径或 base URL
  totalSlides: number
  defaultTransition?: 'fade' | 'slide' | 'zoom' | 'none'
}

export class SlideRunner {
  private current = 0
  private host: HTMLElement
  private shadow: ShadowRoot | null = null

  constructor(private options: SlideRunnerOptions) {
    this.host = options.container
    this.bindKeyboard()
  }

  async navigateTo(index: number, transition?: string): Promise<void> {
    if (index < 0 || index >= this.options.totalSlides) return

    const url = `${this.options.slidesDir}/slide-${String(index + 1).padStart(3, '0')}.html`
    const html = await fetch(url).then(r => r.text())

    // 每次切换重建 Shadow DOM，保证 CSS/JS 完全隔离
    if (this.shadow) this.host.innerHTML = ''
    this.shadow = this.host.attachShadow({ mode: 'open' })
    this.shadow.innerHTML = html

    // 执行 slide 内的 <script> 标签（Shadow DOM 不自动执行）
    this.executeScripts(this.shadow)

    // 入场动画
    await this.playEnterAnimation(this.shadow, transition ?? this.options.defaultTransition)

    this.current = index
  }

  private executeScripts(root: ShadowRoot) {
    root.querySelectorAll('script').forEach(old => {
      const script = document.createElement('script')
      script.textContent = old.textContent
      root.appendChild(script)
      old.remove()
    })
  }

  private async playEnterAnimation(root: ShadowRoot, type = 'fade') {
    const slide = root.querySelector<HTMLElement>('.slide')
    if (!slide) return
    // 依赖全局 GSAP（在 index.html 中加载）
    const gsap = (window as any).gsap
    if (!gsap) return

    const animations: Record<string, object> = {
      fade:  { opacity: 0, duration: 0.4 },
      slide: { x: 60, opacity: 0, duration: 0.45 },
      zoom:  { scale: 0.92, opacity: 0, duration: 0.4 },
      none:  {},
    }
    if (type !== 'none') gsap.from(slide, animations[type] ?? animations.fade)
  }

  next() { this.navigateTo(this.current + 1) }
  prev() { this.navigateTo(this.current - 1) }

  private bindKeyboard() {
    document.addEventListener('keydown', (e) => {
      if (e.key === 'ArrowRight' || e.key === ' ') this.next()
      if (e.key === 'ArrowLeft')                   this.prev()
      if (e.key === 'f' || e.key === 'F')          document.documentElement.requestFullscreen()
    })
  }
}
```

### 3.2 `@tang-slidex/builder` — 构建 & 导出

#### 3.2.1 HTML 单文件内联打包

原理：读取 `index.html` + 所有 `slides/*.html` + `assets/`，将所有外部资源（CSS、JS、图片）内联，生成一个零依赖的 `.html` 文件。

```typescript
// packages/builder/src/inline/InlineBuilder.ts
import { readFileSync, writeFileSync } from 'fs'
import { resolve, extname } from 'path'
import { glob } from 'glob'

export class InlineBuilder {
  constructor(private projectDir: string) {}

  async build(outputPath: string): Promise<void> {
    // 1. 读取所有 slide HTML，转为 Base64 data URL 嵌入 index.html 的 <script> 中
    const slides = await glob(`${this.projectDir}/slides/slide-*.html`)
    const slideMap: Record<string, string> = {}
    for (const f of slides) {
      const name = f.split('/').pop()!
      slideMap[name] = readFileSync(f, 'utf-8')
    }

    // 2. 内联 assets（图片 → Base64，CSS/JS → 内嵌 <style>/<script>）
    const inlinedSlides = await this.inlineAssets(slideMap)

    // 3. 生成 index.html，嵌入 slide 数据
    let index = readFileSync(`${this.projectDir}/index.html`, 'utf-8')
    const slideDataScript = `<script>window.__SLIDES__ = ${JSON.stringify(inlinedSlides)}</script>`
    index = index.replace('</head>', `${slideDataScript}\n</head>`)

    // 4. 替换 fetch 逻辑为从内存读取
    index = index.replace(
      `await fetch(url).then(r => r.text())`,
      `await Promise.resolve(window.__SLIDES__[url.split('/').pop()])`
    )

    writeFileSync(outputPath, index)
    console.log(`✅ 构建完成：${outputPath}`)
  }

  private async inlineAssets(slides: Record<string, string>): Promise<Record<string, string>> {
    // 将每个 slide HTML 中的 <img src="..."> 替换为 Base64 data URI
    const result: Record<string, string> = {}
    for (const [name, html] of Object.entries(slides)) {
      result[name] = html.replace(
        /src="(\.\.\/assets\/[^"]+)"/g,
        (_, src) => {
          const absPath = resolve(this.projectDir, 'slides', src)
          const ext = extname(absPath).slice(1)
          const data = readFileSync(absPath).toString('base64')
          return `src="data:image/${ext};base64,${data}"`
        }
      )
    }
    return result
  }
}
```

#### 3.2.2 Puppeteer PDF 导出

```typescript
// packages/builder/src/pdf/PdfExporter.ts
import puppeteer from 'puppeteer'
import { resolve } from 'path'

export interface PdfExportOptions {
  inputHtml:   string   // presentation.html 绝对路径
  outputPdf:   string   // 输出 PDF 路径
  width:       number   // 1280
  height:      number   // 720
  totalSlides: number
}

export async function exportPdf(opts: PdfExportOptions): Promise<void> {
  const browser = await puppeteer.launch({ headless: true })
  const page    = await browser.newPage()

  await page.setViewport({ width: opts.width, height: opts.height, deviceScaleFactor: 2 })
  await page.goto(`file://${opts.inputHtml}`, { waitUntil: 'networkidle0' })

  const screenshots: Buffer[] = []

  for (let i = 0; i < opts.totalSlides; i++) {
    // 导航到第 i 页
    await page.evaluate((idx) => {
      (window as any).__slideRunner?.navigateTo(idx)
    }, i)

    // 等待动画完成
    await page.waitForTimeout(700)

    const shot = await page.screenshot({ type: 'png' })
    screenshots.push(shot as Buffer)
  }

  await browser.close()

  // 用 sharp 或 pdf-lib 将多张截图合并为 PDF
  await mergeToPdf(screenshots, opts.outputPdf, opts.width, opts.height)
  console.log(`✅ PDF 导出完成：${opts.outputPdf}`)
}

async function mergeToPdf(screenshots: Buffer[], outputPath: string, w: number, h: number) {
  const { PDFDocument } = await import('pdf-lib')
  const doc = await PDFDocument.create()
  for (const shot of screenshots) {
    const img  = await doc.embedPng(shot)
    const page = doc.addPage([w, h])
    page.drawImage(img, { x: 0, y: 0, width: w, height: h })
  }
  const { writeFileSync } = await import('fs')
  writeFileSync(outputPath, await doc.save())
}
```

// packages/builder/src/cli.ts
// 使用方式：
//   tang-slidex build          → 输出 dist/presentation.html

  .option('--pdf', '导出 PDF')

      const manifest = require(`${process.cwd()}/tang-slidex.config.json`)
      await exportPdf({
        inputHtml:   '/tmp/tang-slidex-tmp.html',
        outputPdf:   opts.output,
        width:       1280,
        height:      720,
        totalSlides: manifest.totalSlides,
      })
    }
  })

#### 3.4.3 AI Chat 组件与 Agent 通信

```tsx
// packages/editor/src/renderer/components/AiChat/index.tsx
// Agent 修改 slide 的核心流程：
//
// 用户: "把第 3 页的标题改成蓝色"
//   → 解析意图：目标页 = slide-003.html，操作 = 修改标题颜色
//   → 读取 slide-003.html 内容
//   → 调用 AI（附上文件内容 + 修改指令）
//   → AI 返回修改后的完整 HTML
//   → 写回 slide-003.html
//   → 触发预览热更新

interface Message {
  role:    'user' | 'assistant'
  content: string
}

export function AiChat({ currentSlide, totalSlides, onSlideUpdated }: Props) {
  const [messages, setMessages] = useState<Message[]>([])
  const [input, setInput]       = useState('')
  const [loading, setLoading]   = useState(false)

  async function sendMessage() {
    if (!input.trim()) return
    setLoading(true)

    const userMsg: Message = { role: 'user', content: input }
    setMessages(prev => [...prev, userMsg])
    setInput('')

    // 调用 Electron IPC → main 进程 → 读取文件 + 调用 AI API
    const result = await window.electronAPI.chat({
      messages: [...messages, userMsg],
      currentSlideIndex: currentSlide,
      projectDir: await window.electronAPI.getProjectDir(),
    })

    setMessages(prev => [...prev, { role: 'assistant', content: result.message }])

    if (result.updatedSlides?.length) {
      onSlideUpdated()  // 刷新预览
    }

    setLoading(false)
  }

  return (
    <div className="flex flex-col h-full">
      <div className="px-4 py-3 border-b border-gray-800 font-semibold text-sm">
        AI 助手
      </div>
      {/* 消息列表 */}
      <div className="flex-1 overflow-y-auto p-4 space-y-3">
        {messages.map((m, i) => (
          <MessageBubble key={i} message={m} />
        ))}
        {loading && <TypingIndicator />}
      </div>
      {/* 输入框 */}
      <div className="p-3 border-t border-gray-800 flex gap-2">
        <textarea
          className="flex-1 bg-gray-800 rounded-lg px-3 py-2 text-sm resize-none"
          rows={2}
          value={input}
          onChange={e => setInput(e.target.value)}
          onKeyDown={e => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage() } }}
          placeholder="描述你想要的修改..."
        />
        <button
          onClick={sendMessage}
          disabled={loading}
          className="px-3 py-2 bg-blue-600 rounded-lg text-sm disabled:opacity-50"
        >
          发送
        </button>
      </div>
    </div>
  )
}
```

## 5. Markdown 处理流水线

### 5.1 两级 Markdown 说明

| 文件 | 类型 | 书写者 | 格式 |
|------|------|--------|------|
| `content.md` | 内容草稿 | 用户（自然语言） | 自由 Markdown，无格式限制 |
| `execution.md` | 执行计划 | Agent 生成 | 半结构化（每页有约定字段，但可自然语言补充） |

### 5.2 content.md → execution.md 转换

由 `ppt-outline` Skill 引导 Agent 完成，核心流程：

```
1. Agent 读取 content.md
2. 提取：主题、受众、核心观点、数据点、章节逻辑
3. 若内容模糊 → 向用户提问（反馈机制）
4. 生成 execution.md：
   - 全局 frontmatter（标题、主题、页数、比例）
   - 每页描述块：类型、标题、文案、图表方案、动画、布局
```

### 5.3 execution.md → HTML 转换

由 `slide-generator` Skill 引导 Agent 完成：

```
for 每页描述 in execution.md:
    1. 识别页面类型（cover / toc / content / data-chart / ...）
    2. 选择对应模板结构
    3. 填充文案、数据、图表配置
    4. 编写入场动画
    5. 输出 slide-NNN.html
```

## 8. 编辑器架构

### 8.1 Electron 主进程职责

```typescript
// packages/editor/src/main/index.ts

// 主进程负责：
// 1. 文件系统操作（读写 slide HTML）
// 2. 调用 AI API（保护 API Key 安全，不暴露到渲染进程）
// 3. 启动 Vite 开发服务器（预览热更新）
// 4. 触发 builder 执行构建/导出
// 5. IPC 通信桥

ipcMain.handle('chat', async (_, payload) => {
  const { messages, currentSlideIndex, projectDir } = payload

  // 读取当前 slide 文件内容，加入上下文
  const slidePath  = path.join(projectDir, 'slides', `slide-${String(currentSlideIndex + 1).padStart(3,'0')}.html`)
  const slideContent = fs.readFileSync(slidePath, 'utf-8')

  // 构建 AI 请求（含文件内容作为上下文）
  const response = await callAI([
    ...messages,
    { role: 'system', content: `当前编辑的文件是 ${slidePath}，内容如下：\n\n${slideContent}` }
  ])

  // 如果 AI 返回了修改后的 HTML，写回文件
  if (response.updatedHtml) {
    fs.writeFileSync(slidePath, response.updatedHtml)
    return { message: response.message, updatedSlides: [currentSlideIndex] }
  }

  return { message: response.message }
})
```

### 8.2 预览热更新

```typescript
// 方案：Electron webContents.reload()
// 当 slides/*.html 文件变化时，通知渲染进程刷新预览

import chokidar from 'chokidar'

const watcher = chokidar.watch(`${projectDir}/slides/*.html`)
watcher.on('change', (filePath) => {
  const slideIndex = parseSlideIndex(filePath)  // slide-003.html → 2
  mainWindow.webContents.send('slide-updated', slideIndex)
})

// 渲染进程监听
window.electronAPI.onSlideUpdated((slideIndex: number) => {
  if (slideIndex === currentIndex) {
    // 重新加载当前 slide
    runner.navigateTo(slideIndex)
  }
})
```
